describe("canvasToPicture", () => {
  function mockCanvas(
    blobData: Uint8Array | null,
    blobType = "image/jpeg",
  ): HTMLCanvasElement {
    return {
      toBlob(callback: BlobCallback, type?: string, quality?: number) {
        if (blobData === null) {
          callback(null);
          return;
        }
        const blob = new Blob([blobData], { type: type ?? blobType });
        callback(blob);
      },
    } as unknown as HTMLCanvasElement;
  }

  it("should convert canvas to Picture with default options", async () => {
    const imageData = new Uint8Array([0x89, 0x50, 0x4E, 0x47]);
    const canvas = mockCanvas(imageData);
    const result = await canvasToPicture(canvas);

    assertEquals(result.mimeType, "image/jpeg");
    assertEquals(result.type, "FrontCover");
    assertEquals(result.data, imageData);
  });

  it("should use provided format and type options", async () => {
    const canvas = mockCanvas(new Uint8Array([1, 2, 3]));
    const result = await canvasToPicture(canvas, {
      format: "image/png",
      type: "BackCover",
      description: "Back cover",
    });

    assertEquals(result.mimeType, "image/png");
    assertEquals(result.type, "BackCover");
    assertEquals(result.description, "Back cover");
  });

  it("should reject when canvas.toBlob returns null", async () => {
    const canvas = mockCanvas(null);
    let threw = false;
    try {
      await canvasToPicture(canvas);
    } catch (e) {
      threw = true;
      assertEquals((e as Error).message, "Failed to convert canvas to blob");

    assertEquals(threw, true);
  });
